   1） function isPalindrome(line) {
        line += "";
        return line === line.split("").reverse().join("");
    }
   2） function isPalindrome(line) {
        line += "";
        for(var i=0,j=line.length-1;i< j;i++,j--){
            if(line.charAt(i) !== line.charAt(j)){
                return false;
            }
        }
        return true;
    }
